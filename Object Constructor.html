Object Constructor

The object constructor gives us a skeleton of an object that can be used over and over. Each usage creates a new instance of the object and

function Person(first , last, age, eyes ) {
this.firstname = first;
this.lastname = last;
this.age = age;
this.eyes = eyes;
}
var myFather = new Person("John", "Doe", 50, "blue");
var myMother = new Person("Sally", "Rally", 48, "green");

Object constructors can add the use of methods just as in the basic object as well.

function Person(first , last, age, eyes ) {
this.firstname = first;
this.lastname = last;
this.age = age;
this.eyes = eyes;
this.talk = function(){
console.log ('hello ' + this.firstname );
};
}
var myFather = new Person("John", "Doe", 50, "blue");
var myMother = new Person("Sally", "Rally", 48, "green");
