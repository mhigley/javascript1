Strings

The  String global object is a constructor for strings, or a sequence of characters.

.length

The length property represents the length of a string.

console.log( 'matthew'.length);
console.log( str.length); // 7
var fullName = 'Matthew David Higley' ;
console.log( fullName.length); // 20

.charAt()

Returns the character at the specified index (position)

var fullName = 'Matthew David Higley' ;
console.log( fullName.charAt()); // M
console.log( fullName.charAt(8)) ; // D
console.log( fullName.charAt(250)) ; // ""

.indexOf()

Returns the position of the first found occurrence of a specified value in a string

var fullName = 'Matthew David Higley' ;
console.log( fullName.indexOf('t')) ; // 2
console.log( fullName.indexOf('D')) ; // 8
console.log( fullName.indexOf('H')) ; // 14

.concat()

Combines the text of two strings and returns a new string.

var hello = 'Hello' ;
console.log( hello.concat(' JavaScript')) ; // Hello JavaScript

Tip: For performance reasons, it is recommended to use the assignment operators (+, +=) instead of the concat() method.

var hello = 'Hello' ;
console.log( hello + ' JavaScript'); // Hello JavaScript
hello += ' there';
console.log( hello + ' Javascript'); // Hello there JavaScript

.substring() & .substr()

Extracts the characters from a string

.substring()

Returns a subset of a string between one index and another, or through the end of the string.

var myName = 'Matthew' ;
console.log( myName.substring(0, 3)); // Mat
console.log( myName.substring(3, 6)); // the
console.log( myName.substring(3)) ; // thew

.substr()

Returns the characters in a string beginning at the specified location through the specified number of characters. If length is omitted, substr() extracts characters to the end of the string.

var myName = 'Matthew' ;
console.log( myName.substr(0, 3)); // Mat
console.log( myName.substr(3, 2)); // th
console.log( myName.substr(3)) ; // thew

.toLowerCase() & .toUpperCase()

Returns the calling string value converted to declared casing.

.toLowerCase()

Converts to lowercase.

var myName = 'Matthew' ;
console.log( myName.toLowerCase()); // matthew
var myOtherName = 'DAVID' ;
console.log( myOtherName.toLowerCase()); // david

.toUpperCase()

Converts to uppercase.

var myName = 'Matthew' ;
console.log( myName.toUpperCase()); // MATTHEW
var myOtherName = 'david' ;
console.log( myOtherName.toUpperCase()); // DAVID

.trim()

Removes whitespace from both ends of a string.

var extraSpaces = ' stuff and things ' ;
console.log( extraSpaces);
console.log( extraSpaces.trim());
