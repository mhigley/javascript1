DOM Manipulation

.textContent

The textContent property returns or sets the text from the selected element.

<!-- HTML -->
<div id="set"> </div>
<div id="get"> <p>This is a < strong>different</ strong> line of text.</ p></div >

Setting content

var setDiv = document .querySelector('div#set');
setDiv.textContent = '<p>Here is a <strong>line</strong> of text</p>';

Getting content

var getDiv = document .querySelector('div#get');
console.log (getDiv.textContent); //This is a different line of text.

.innerHTML

The innerHTML property sets or returns the HTML content (inner HTML) of an element.

<!-- HTML -->
<div id="set"> </div>
<div id="get"> <p>This is a < strong>different</ strong> line of text.</ p></div >

Setting content

var setDiv = document .querySelector('div#set');
setDiv.innerHTML = '<p>Here is a <strong>line</strong> of text</p>';

Getting content

var getDiv = document .querySelector('div#get');
console.log (getDiv.innerHTML); //<p>This is a <strong>different</strong> line of text.</p>

document.write()

This method is mostly used for testing purposes. The write() method writes HTML expressions or JavaScript code to a document.

Tip: If document.write() is used after an HTML document is fully loaded, it will delete all existing HTML.

<!-- HTML -->
<h1> Headline</h1>
<p> Lorem ipsum dolor sit amet, consectetur adipisicing elit. Dolores a, nobis quia mollitia. Natus, qui cupiditate eligendi recusandae in, libero!</ p>
<button> Click Here!</button>
/* JavaScript */
var writeMessage = '<h1>Info</h1><p>This content will be loaded onto the page</p>' ;
document.write (writeMessage);
var buttonMessage = '<h1>Note</h1><p>All HTML has been deleted.</p>' ;
document.querySelector( 'button').addEventListener('click' , function(){
document.write (buttonMessage);
});
