Creating Elements

.createElement()

The createElement() method creates an Element Node with the specified name.

var body = document .body;
console.log (body);
var main = document .createElement('main') ,
header = document. createElement('header'),
article = document. createElement('article'),
aside = document. createElement('aside'),
footer = document. createElement('footer');
main.append({ header, article, aside , footer});
body.append( main);
main.className = 'container';

.appendChild

The appendChild() method appends a node as the last child of a node.

var body = document .body;
var main = document .createElement('main') ,
header = document. createElement('header'),
article = document. createElement('article'),
aside = document. createElement('aside'),
footer = document. createElement('footer');
main.appendChild( header);
main.appendChild( article);
main.appendChild( aside);
main.appendChild( footer);
main.className = 'container';
body.appendChild( main);

.append()

The appendChild() method appends a set of nodes after last child of a ParentNode.

var body = document .body;
var main = document .createElement('main') ,
header = document. createElement('header'),
article = document. createElement('article'),
aside = document. createElement('aside'),
footer = document. createElement('footer');
main.append( header, article, aside , footer);
main.className = 'container';
body.append( main);

.insertBefore()

The insertBefore() method inserts a node as a child, right before an existing child, which you specify.

<!-- HTML -->
<div id="parent">
<p id="child" >Lorem ipsum dolor sit amet.</ p >
</div>
var h1 = document .createElement("h1") ;
// Get a reference to the element, before we want to insert the element
var para = document .getElementById("child") ;
// Get a reference to the parent element
var parent = para .parentNode;
// Insert the new element into the DOM before sp2
parent.insertBefore (h1, para);

Adding Attributes

Adding attibutes and their values are as easy as using dot notation.

.id

var p = document .createElement('p') ;
p.id = 'paragraphId';
p.innerHTML = 'This is my <strong>amazing</strong> paragraph';
parent.appendChild( p);

.className

var h1 = document .createElement('h1') ;
h1.className = 'headingClass';
h1.textContent = 'The headline to an article';
parent.appendChild( h1);

.alt

var img = document .createElement('img') ;
img.setAttribute( 'alt', 'Picture of a cute, cuddly raptor.');
img.src = 'http://vignette3.wikia.nocookie.net/dino/images/5/5a/Jurassic_Park_Baby_Raptor_Hatchling_1_1_Replica/revision/latest?cb=20151114025025';
parent.appendChild( img);
